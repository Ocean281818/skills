#### 提取视频数据的JavaScript代码
```javascript
// 在browser_console中执行
const videos = [];
const items = document.querySelectorAll('a[href*="video"]');
const seen = new Set();
items.forEach(item => {
  const href = item.href;
  if (!href.includes('bilibili.com/video') || seen.has(href)) return;
  seen.add(href);
  const title = item.getAttribute('title') || item.textContent?.trim().split('\n')[0];
  if (title && title.length > 5) {
    const parent = item.closest('div') || item.parentElement;
    videos.push({ title: title.substring(0, 80), href, info: parent?.innerText?.substring(0, 200) });
  }
});
JSON.stringify(videos.slice(0, 30), null, 2);
```
